# Herencia:
# La herencia es una propiedad de la programación orientada a objetos
# que permite que una clase (llamada clase hija o subclase) adopte 
# atributos y métodos de otra clase (llamada clase padre o superclase). 
# Esto facilita la reutilización de código, permite organizar mejor la 
# estructura del programa y da la posibilidad de extender o modificar 
# el comportamiento de la clase padre en la clase hija.

# Ejemplo de herencia:

# Clase padre
class Animal:
    def __init__(self, nombre, edad):
        self.nombre = nombre
        self.edad = edad

    def hablar(self):
        print("Este animal hace un sonido.")

# Clase hija que hereda de Animal
class Perro(Animal):
    def hablar(self):
        print(f"{self.nombre} dice: ¡Guau!")

class Gato(Animal):
    def hablar(self):
        print(f"{self.nombre} dice: ¡Miau!")

# Uso de las clases
perro1 = Perro("Max", 3)
gato1 = Gato("Luna", 2)

perro1.hablar()  # Max dice: ¡Guau!
gato1.hablar()   # Luna dice: ¡Miau!
