# La encapsulación es una técnica de la programación orientada a objetos
# que permite proteger los datos dentro de una clase, evitando que otros
# objetos los modifiquen directamente.

# Ejemplo en Python:

class Persona:

    def __init__(self, nombre, edad):
        # Atributos privados (no accesibles directamente desde fuera de la clase)
        self.__nombre = nombre
        self.__edad = edad

    # Método getter para obtener el nombre
    def get_nombre(self):
        return self.__nombre

    # Método setter para modificar el nombre de forma segura
    def set_nombre(self, nuevo_nombre):
        self.__nombre = nuevo_nombre


# Crear un objeto Persona
persona = Persona("Ana", 20)

# Obtener el nombre mediante el getter
print(persona.get_nombre())   # Salida: Ana

# Cambiar el nombre mediante el setter
persona.set_nombre("Beatriz")

# Mostrar el nombre modificado
print(persona.get_nombre())   # Salida: Beatriz
